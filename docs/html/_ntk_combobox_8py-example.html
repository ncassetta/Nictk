29 def change_label(event):
30  """This is called when you select a city in the upper combo;
31  it prints the selected city on the helper label."""
32  sel = cmbCities.get_content()
33  labHelp.set_content("You selected: " + sel)

36 def delete_sel(event):
37  """This is called by ButDel; it deletes the selected city
38  from the upper combo and adds it to the lower."""
39  # gets the selected string
40  sel = cmbCities.get_content()
41  if len(sel):
42  # gets its numeric index (from 0)
43  ind = cmbCities.index(sel)
44  # deletes from upper combo (cmbCities.delete(sel) also works)
45  cmbCities.delete(ind)
46  # adds the string to the lower combo
47  cmbIns.add(sel)
48  if cmbCities.get_menu().size() > 0:
49  # selects the next city in the upper combobox (if sel was the last string
50  # the menu auto adjust the index, so no out of range errors)
51  cmbCities.set_content(cmbCities.get_item(ind))
52  labHelp.set_content("You deleted {} (but it is now in the lower combobox".format(sel))
53  else:
54  # the upper combo is empty
55  cmbCities.set_content("")
56  labHelp.set_content("Empty menu")
57 
58 
59 def insert_sel(event):
60  """This is called by ButIns; it deletes a city in the lower combo
61  and adds it to the upper."""
62  sel = cmbIns.get_content()
63  if len(sel):
64  # see delete_sel
65  ind = cmbIns.index(sel)
66  cmbCities.add(sel)
67  cmbIns.delete(sel)
68  labHelp.set_content("You newly added {} to the upper combobox".format(sel))
69  if cmbIns.get_menu().size() > 0:
70  cmbIns.set_content(cmbIns.get_item(ind))
71  else:
72  cmbIns.set_content("")
73 
74 
75 def reset_combo(event):
76  """This callback resets the contents of the two combos"""
77  # empties and refills the upper combo
78  cmbCities.delete(0, END)
79  for i in CITIES:
80  cmbCities.add(i)
81  # selects the first city
82  cmbCities.set_content(CITIES[0])
83  # empties the lower combo
84  cmbIns.delete(0, END)
85  cmbIns.set_content("")
86  labHelp.set_content("You reset the combos")

89 winMain = Ntk.NtkMain(200, 150, 400, 300, "Combobox sample")
90 # configures all children widget when they are created
91 winMain.config_children(ALL, font=("Arial", 16))
92 winMain.config_children((Ntk.NtkButton, Ntk.NtkCombobox), bcolor="#D0D080", abcolor="#E0E090")
93 # creates the combobox with the cities
94 cmbCities = Ntk.NtkCombobox(winMain, 0, 0, FILL, 70, pad=(10, 20, 10, 10),
95  items=CITIES, command=change_label)
96 # creates the label under it
97 labHelp = Ntk.NtkLabel(winMain, 0, PACK, FILL, 90, pad=10,
98  content="Initially selected: " + CITIES[0])
99 labHelp.config(bcolor="#B0D0F0", relief=RIDGE, anchor=CENTER)
100 # rowframe for aligning other widgets
101 rfr1 = Ntk.NtkRowFrame(winMain, 0, PACK, FILL, FILL)
102 rfr1.add_row(55)
103 # button for deleting a city from cmbCities
104 butDel = Ntk.NtkButton(rfr1, 0, 0, "50%", FILL, pad=(10, 10, 10, 5),
105  content="Delete selected", command=delete_sel)
106 # button for reset
107 butRes = Ntk.NtkButton(rfr1, PACK, 0, FILL, FILL, pad=(10, 10, 10, 5),
108  content="Reset", command=reset_combo)
109 rfr1.add_row(55)
110 # button for reinserting a deleted city
111 butIns = Ntk.NtkButton(rfr1, 0, 0, "50%", FILL, pad=(10, 5, 10, 10),
112  content="Insert selected", command=insert_sel)
113 butIns.deactivate()
114 # combobox for deleted cities
115 cmbIns = Ntk.NtkCombobox(rfr1, PACK, 0, FILL, FILL, pad=(10, 5, 10, 10))
116 # auto activate - deactivate buttons
117 cmbCities.bind("<<ChangedVar>>",
118  lambda ev: butDel.activate() if len(ev.widget.get_content()) else butDel.deactivate())
119 cmbIns.bind("<<ChangedVar>>",
120  lambda ev: butIns.activate() if len(ev.widget.get_content()) else butIns.deactivate())
