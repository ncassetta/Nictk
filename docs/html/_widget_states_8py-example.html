20import _setup
21
22import Nictk as Ntk
23from Nictk.constants import *
24from time import *
25
26bcolors = ("#FFFFFF", "#202060", "#D0D0A0", "#206020", "#FFA0A0", "#402000")
27fcolors = ("#000000", "#FFFFC0", "#002060", "#FFFFC0", "#004040", "#D0B0FF")
28dfcolors = ("#202020", "#D0D0A0", "#204080", "#D0D0A0", "#206060", "#A0A0D0")
29explstates = ("Now the buttons and the entry are enabled: you can type into the entry and press the buttons",
30 "Now the buttons and the entry are disabled: you cannot type into the entry or press the buttons",
31 "Now the buttons and the entry are hidden")
32texts = ("If you change the state of a container ...", "... all of its children inherits it")
33butcaptions = ("Disable", "Hide", "Show")
34colorind, textind = 0, 0
35
36
37def change_color(event):
38 """This is called by butColor and changes the colors for the entText entry."""
39 global colorind
40 colorind = (colorind + 1) % len(bcolors)
41 entText.config(bcolor = bcolors[colorind], fcolor = fcolors[colorind], dfcolor= dfcolors[colorind])
42
43def change_text(event):
44 """This is called by butFeame and changes the text of labFrame."""
45 global textind
46 textind = (textind + 1) % 2
47 labFrame.set_content(texts[textind])
48
49def change_status(event):
50 """This is called by butHideShow and ciclicaly disables and hides
51 butColor, entText and hfr1."""
52 # event.widget.get_content() also works
53 txt = butHideShow.get_content()
54 if txt == "Disable":
55 entText.deactivate()
56 butColor.deactivate()
57 hfr1.deactivate()
58 newind = 1
59 elif txt == "Hide":
60 entText.hide()
61 butColor.hide()
62 hfr1.hide()
63 newind = 2
64 elif txt == "Show":
65 entText.show()
66 entText.activate()
67 butColor.show()
68 butColor.activate()
69 hfr1.show()
70 hfr1.activate()
71 newind = 0
72 butHideShow.set_content(butcaptions[newind])
73 labExplain.set_content(explstates[newind])
74
75
76
77winMain = Ntk.Main(200, 150, 400, 300, "Widget states sample")
78# widgets are aligned with absolute coords
79
80# upper entry and button
81entText = Ntk.Entry(winMain, 0, 0, "fill", 60, pad=10)
82entText.config(font=("Arial", 14), bcolor=bcolors[0], fcolor=fcolors[0], dfcolor = dfcolors[0])
83
84butColor = Ntk.Button(winMain, CENTER, 70, 100, 30,
85 content="Change Color", command=change_color)
86
87# frame and its children
88hfr1 = Ntk.HorFrame(winMain, 0, 110, FILL, 80, content="This is a HorFrame")
89hfr1.config(relief=RIDGE)
90labFrame= Ntk.Label(hfr1, 0, 0, "70%", FILL, pad=(8, 8, 8, 28), content=texts[0])
91labFrame.config(fcolor="#FFFFC0", bcolor="#202060", dfcolor="#D0D0A0",
92 relief=RIDGE)
93butFrame= Ntk.Button(hfr1, PACK, 0, FILL, FILL, pad=(8, 8, 8, 28), content=
94 "Click me", command=change_text)
95
96# lower button and label
97butHideShow = Ntk.Button(winMain, 10, 220, 110, 50,
98 content=butcaptions[0], command=change_status)
99butHideShow.config(bcolor="#C0C0FF")
100labExplain = Ntk.Label(winMain, 130, 220, -10, 50)
101labExplain.config(bcolor="#C0C0FF")
102labExplain.set_content(explstates[0])
103
104Ntk.mainloop()
