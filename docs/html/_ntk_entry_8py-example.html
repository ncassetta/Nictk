24 def get_focus(event):
25  """This is called when you click on one of the two entries."""
26  if event.widget == entCelsius:
27  # the Celsius entry got the focus ...
28  entCelsius.bind("<Return>", to_Farenheit)
29  entCelsius.bind("<<ChangedVar>>", to_Farenheit)
30  entFarenheit.unbind("<<ChangedVar>>")
31  else:
32  # if the Farenheit entry got the focus ...
33  entFarenheit.bind("<Return>", to_Celsius)
34  entFarenheit.bind("<<ChangedVar>>", to_Celsius)
35  entCelsius.unbind("<<ChangedVar>>")
36 
37 def to_Farenheit(event):
38  """Conversion F -> C"""
39  try:
40  cel = float(entCelsius.get_content())
41  far = "{:.2f}".format(cel * 1.8 + 32)
42  except ValueError:
43  far=""
44  entFarenheit.set_content(far)

46 def to_Celsius(event):
47  """Conversion C -> F"""
48  try:
49  far = float(entFarenheit.get_content())
50  cel = "{:.2f}".format((far - 32) / 1.8)
51  except ValueError:
52  cel = ""
53  entCelsius.set_content(cel)
